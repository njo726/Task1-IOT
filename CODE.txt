#include <WiFi.h>
#include <HTTPClient.h>

const char* ssid = "Wokwi-GUEST";
const char* password = "";

const String url = "https://raw.githubusercontent.com/njo726/Task1-IOT/main/LEFT.html";
String payload = "";

// Define LED pins
const int ledPinF1 = 33;  // LED for RIGHT
const int ledPinF2 = 25;  // LED for UP
const int ledPinS1 = 26;  // LED for DOWN
const int ledPinL1 = 27;  // LED for LEFT

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  pinMode(ledPinF1, OUTPUT);
  pinMode(ledPinF2, OUTPUT);
  pinMode(ledPinS1, OUTPUT);
  pinMode(ledPinL1, OUTPUT);

  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
 }

  Serial.print("OK! IP=");
  Serial.println(WiFi.localIP());

  Serial.print("Fetching " + url + "... ");
}

void turnOffAllLEDs() {
  digitalWrite(ledPinF1, LOW);
  digitalWrite(ledPinF2, LOW);
  digitalWrite(ledPinS1, LOW);
  digitalWrite(ledPinL1, LOW);
}

void loop() {
  HTTPClient http;
  http.begin(url);
  int httpResponseCode = http.GET();
  if (httpResponseCode > 0) {
    Serial.print("HTTP ");
    Serial.println(httpResponseCode);
    payload = http.getString();
    Serial.println();
    Serial.println(payload);
    
    if (payload == "RIGHT") {
 // Turn on the LEDs for "RIGHT"
      digitalWrite(ledPinF1, HIGH);
      digitalWrite(ledPinF2, LOW);
      digitalWrite(ledPinS1, LOW);
      digitalWrite(ledPinL1, LOW);
    } else if (payload == "UP") {
      // Turn on the LED for "UP"
      digitalWrite(ledPinF1, LOW);
      digitalWrite(ledPinF2, HIGH);
      digitalWrite(ledPinS1, LOW);
      digitalWrite(ledPinL1, LOW);
      } else if (payload == "DOWN") {
      // Turn on the LED for "DOWN"
      digitalWrite(ledPinF1, LOW);
      digitalWrite(ledPinF2, LOW);
      digitalWrite(ledPinS1, HIGH);
      digitalWrite(ledPinL1, LOW);
    } else if (payload == "LEFT") {
      // Turn on the LED for "Left"
      digitalWrite(ledPinF1, LOW);
      digitalWrite(ledPinF2, LOW);
      digitalWrite(ledPinS1, LOW);
      digitalWrite(ledPinL1, HIGH);
    } else {
      // Turn off all LEDs for unrecognized letters or other responses
      turnOffAllLEDs();
    }
 } else {
    Serial.print("Error code: ");
    Serial.println(httpResponseCode);
    Serial.println(":-(");
  }
  
  http.end();
  delay(1000); // Wait for 1 second before making the next HTTP request
}
